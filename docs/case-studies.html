14.3 Case Study 1: UNF Basketball Attendance Analysis

14.3.1 Problem Definition

The UNF Athletics Director wants to understand what drives attendance at basketball games and identify strategies to increase fan engagement. Specifically, they want to know:

  1. What factors are associated with higher attendance?
  2. Can we predict attendance for upcoming games?
  3. What marketing strategies should we prioritize?

14.3.2 Data Collection and Import

We’ll create a dataset representing two seasons of UNF basketball games with various factors that might influence attendance.

library(dplyr)
library(ggplot2)
library(caret)
library(lubridate)
library(patchwork)

# Create comprehensive attendance dataset
set.seed(456)
n_games <- 50

attendance_data <- data.frame(
  game_id = 1:n_games,
  date = seq(as.Date("2022-11-01"), by = "week", length.out = n_games),
  opponent = sample(c("Top-Ranked", "Conference", "Non-Conference", "Rival"),
                   n_games, replace = TRUE,
                   prob = c(0.15, 0.4, 0.3, 0.15)),
  day_of_week = sample(c("Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"),
                      n_games, replace = TRUE,
                      prob = c(0.15, 0.15, 0.2, 0.25, 0.25)),
  ticket_promotion = sample(c("Yes", "No"), n_games, replace = TRUE, prob = c(0.3, 0.7)),
  giveaway = sample(c("Yes", "No"), n_games, replace = TRUE, prob = c(0.25, 0.75)),
  weather = sample(c("Clear", "Rain", "Cold"), n_games, replace = TRUE,
                  prob = c(0.6, 0.25, 0.15)),
  team_win_streak = sample(0:7, n_games, replace = TRUE),
  exam_week = sample(c("Yes", "No"), n_games, replace = TRUE, prob = c(0.15, 0.85))
)

# Generate attendance based on realistic factors
attendance_data <- attendance_data %>%
  mutate(
    # Base attendance
    base_attendance = 1200,

    # Opponent effect
    opponent_effect = case_when(
      opponent == "Rival" ~ 800,
      opponent == "Top-Ranked" ~ 600,
      opponent == "Conference" ~ 300,
      opponent == "Non-Conference" ~ 100
    ),

    # Day of week effect
    day_effect = case_when(
      day_of_week == "Saturday" ~ 400,
      day_of_week == "Friday" ~ 200,
      day_of_week == "Thursday" ~ 50,
      TRUE ~ -100
    ),

    # Promotion effects
    promo_effect = ifelse(ticket_promotion == "Yes", 250, 0),
    giveaway_effect = ifelse(giveaway == "Yes", 180, 0),

    # Weather effect
    weather_effect = case_when(
      weather == "Clear" ~ 100,
      weather == "Rain" ~ -150,
      weather == "Cold" ~ -50
    ),

    # Win streak effect (each win adds fans)
    streak_effect = team_win_streak * 80,

    # Exam week penalty
    exam_effect = ifelse(exam_week == "Yes", -300, 0),

    # Calculate total attendance with random noise
    attendance = pmax(500,  # minimum attendance
                     base_attendance + opponent_effect + day_effect +
                     promo_effect + giveaway_effect + weather_effect +
                     streak_effect + exam_effect + rnorm(n_games, 0, 150))
  ) %>%
  dplyr::select(game_id, date, opponent, day_of_week, ticket_promotion,
                giveaway, weather, team_win_streak, exam_week, attendance)
