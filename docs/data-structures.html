3.2 What Is a Data Frame?

A data frame is R’s version of a spreadsheet—a table with rows and columns where:

  • Each column is a variable (like “Team”, “Points”, “Wins”)
  • Each row is an observation (like one game or one player)
  • Columns can be different types (numeric, character, etc.)
  • All columns must have the same length

Think of it like this:

     Team        Points  Rebounds  Assists
1    UNF           78       42       18
2    FGCU          82       38       22
3    Jacksonville  71       45       15
4    Stetson       88       40       25

3.2.1 Why Data Frames?

Data frames are perfect for sport analytics because:

  1. Structured - Organized like a spreadsheet
  2. Mixed types - Can have text, numbers, TRUE/FALSE in different columns
  3. Powerful - Built-in functions for analysis
  4. Standard - Most R packages expect data frames
  5. Familiar - Works like Excel, but more powerful

# Read CSV file
data <- read.csv("path/to/your/file.csv")

data <- read.csv("file.csv",

3.6 Filtering Data Frames

3.6.1 Filtering Rows by Condition

Use logical conditions to filter rows:

# Create larger dataset
team_stats <- data.frame(
  team = c("UNF", "FGCU", "Jacksonville", "Stetson", "Liberty", "Kennesaw"),
  wins = c(15, 18, 12, 20, 16, 14),
  losses = c(5, 7, 8, 5, 9, 11),
  points_per_game = c(78.5, 82.3, 71.2, 85.1, 76.8, 74.3)
)

# Teams with 15+ wins
team_stats[team_stats$wins >= 15, ]
##      team wins losses points_per_game
## 1     UNF   15      5            78.5
## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1
## 5 Liberty   16      9            76.8
# Teams with winning percentage > .700
winning_teams <- team_stats[team_stats$wins / (team_stats$wins + team_stats$losses) > 0.7, ]
winning_teams
##      team wins losses points_per_game
## 1     UNF   15      5            78.5
## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1
# Teams scoring over 80 ppg
team_stats[team_stats$points_per_game > 80, ]
##      team wins losses points_per_game
## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1

3.6.2 Multiple Conditions

# AND: Both conditions must be TRUE
team_stats[team_stats$wins >= 15 & team_stats$points_per_game > 75, ]
##      team wins losses points_per_game
## 1     UNF   15      5            78.5
## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1
## 5 Liberty   16      9            76.8
# OR: Either condition can be TRUE
team_stats[team_stats$wins >= 18 | team_stats$points_per_game >= 85, ]
##      team wins losses points_per_game
## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1
# NOT: Reverse a condition
team_stats[!(team_stats$wins < 15), ]  # Same as wins >= 15
##      team wins losses points_per_game

## 2    FGCU   18      7            82.3
## 4 Stetson   20      5            85.1
## 5 Liberty   16      9            76.8

# Teams with 15+ wins

subset(team_stats, wins >= 15, select = c(team, wins, points_per_game))

3.11 Real-World Example: Season Analysis

Let’s put it all together:

# Create a season dataset
season <- data.frame(
  game = 1:10,
  opponent = c("FGCU", "Jacksonville", "Stetson", "Liberty",
               "Kennesaw", "FGCU", "Jacksonville", "Stetson",
               "Liberty", "Kennesaw"),
  home = c(TRUE, TRUE, FALSE, TRUE, FALSE,
           FALSE, FALSE, TRUE, TRUE, FALSE),
  points_for = c(78, 82, 71, 88, 76, 85, 79, 91, 74, 83),
  points_against = c(71, 79, 73, 82, 80, 81, 77, 88, 76, 80)
)

# Analysis using dplyr
season_analysis <- season %>%
  # Calculate win/loss and margin
  mutate(
    margin = points_for - points_against,
    result = ifelse(margin > 0, "W", "L"),
    close_game = abs(margin) <= 5
  ) %>%
  # Add game type
  mutate(
    game_type = ifelse(home, "Home", "Away")
  )

# Overall record
season_analysis %>%
  summarise(
    games = n(),
    wins = sum(result == "W"),
    losses = sum(result == "L"),
    win_pct = mean(result == "W") * 100,
    avg_margin = mean(margin)
  )
##   games wins losses win_pct avg_margin
## 1    10    7      3      70          2
# Home vs Away splits
season_analysis %>%
  group_by(game_type) %>%
  summarise(
    games = n(),
    wins = sum(result == "W"),
    avg_points = mean(points_for),
    avg_allowed = mean(points_against)
  )
## # A tibble: 2 × 5
##   game_type games  wins avg_points avg_allowed
##   <chr>     <int> <int>      <dbl>       <dbl>
## 1 Away          5     3       78.8        78.2
## 2 Home          5     4       82.6        79.2
# Performance by opponent
season_analysis %>%
  group_by(opponent) %>%
  summarise(

    wins = sum(result == "W"),
    avg_margin = mean(margin)
  ) %>%
  arrange(desc(wins), desc(avg_margin))
